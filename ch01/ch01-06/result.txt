#define
EOF
(-1)

1

0

int
main(int
argc,
char
const
*argv[])
{
int
status
,
c;
status
=
OUT;
while
((c
=

{
if
(c
==
'
'
||
c
==
'\t'
||
c
==
'\n')
{
if
(status
==
IN)
{
//
空白字符开始
putchar('\n');
status
=
OUT;
}
//
空白字符时不打印
//
putchar(c);
}
else
{
if
(status
==
OUT)
{
//
非空白字符（单词）开始
status
=
IN;
}
putchar(c);
}
}
return
0;
}
